单次抽奖逻辑 (singleDraw)
这是抽奖系统的核心函数，根据权重随机选择物品，并处理史诗物品的保底机制。
/** * 执行单次抽奖操作的核心函数 * 基于权重随机算法选择奖励物品，并处理周期重置和史诗物品保底机制 * @returns {string} 抽中的物品名称 */ function singleDraw() { // 周期检查：每20000抽为一个完整周期，达到后重置奖池 if (state.totalDraws >= 20000) { // 重置抽奖周期和史诗物品剩余数量 state.totalDraws = 0; state.epicRewardsRemaining = {...epicRewardsQuota}; updateEpicRewardsTable(); // 更新UI显示 } // 增加总抽奖次数计数 state.totalDraws++; // 获取当前有效的权重配置（考虑史诗物品保底限制） const { weights, totalWeight } = getCurrentWeights(); // 生成随机数决定抽奖结果（基于总权重的随机分布） const random = Math.random() * totalWeight; // 遍历权重表，确定最终抽中的物品 let currentWeight = 0; let selectedItem = null; for (const [name, weight] of Object.entries(weights)) { currentWeight += weight; // 当随机数落在当前物品的权重区间内时，选中该物品 if (random <= currentWeight) { selectedItem = allItems.find(item => item.name === name); break; } } // 史诗物品保底处理：如果抽中史诗物品且该物品还有剩余数量 if (selectedItem.category === "史诗" && state.epicRewardsRemaining[selectedItem.name] > 0) { // 减少该史诗物品的剩余数量 state.epicRewardsRemaining[selectedItem.name]--; // 更新UI显示 updateEpicRewardsTable(); } // 更新进度条显示 updateProgressBar(); // 返回抽中的物品名称 return selectedItem.name; }
系统对抗关键函数 (autoDrawOnce)
这是系统自动抽奖的核心函数，模拟其他玩家与用户竞争奖励。
/** * 系统自动抽奖函数 - 模拟其他玩家与用户竞争稀有奖励 * 根据当前选择的模式确定抽奖次数，执行批量抽奖并记录结果 */ function autoDrawOnce() { // 根据当前模式确定单次系统抽奖的次数 // 模式映射：1=手动, 2=50抽/5s, 3=500抽/5s, 4=5000抽/5s const count = [null, null, 50, 500, 5000][state.currentMode]; // 重置系统单次统计，准备记录本次抽奖结果 state.systemSingleStats = initializeStats(); // 执行批量抽奖（系统抽奖，isUserDraw=false） const res = batchDraw(count, false); // 检查本次抽奖结果，判断是否触发截胡提示 for (let n in res) { checkEpicPopup(n, false); // 第二个参数false表示这是系统抽奖 } // 从抽奖结果中筛选出史诗物品 const epic = Object.entries(res) .filter(([n]) => allItems.find(it => it.name === n)?.category === '史诗') .map(([n, c]) => `${n}×${c}`); // 格式化显示：物品名称×数量 // 记录系统抽奖日志 if (epic.length) { // 如果抽中了史诗物品，记录详细信息 addSystemLog(`本次抽取获得史诗奖励: ${epic.join(', ')}`); } else { // 如果未抽中史诗物品，记录普通日志 addSystemLog('本次抽取未获得史诗奖励'); } }
系统对抗设计特点
竞争机制：系统模拟其他玩家与用户竞争稀有奖励，增加真实感。
截胡提示：当系统抽到顶级史诗时，会显示"你被截胡了"的提示。
多模式选择：提供不同强度的系统对抗模式。
实时日志：系统抽奖结果会实时显示在日志中。
